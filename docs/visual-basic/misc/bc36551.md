## <a name="example"></a><span data-ttu-id="374ff-108">Przykład</span><span class="sxs-lookup"><span data-stu-id="374ff-108">Example</span></span>

Namespace StringUtility  
    <Extension()> _  
    Module StringExtensions  
        <Extension()> _

        End Sub  
    End Module  
End Namespace
