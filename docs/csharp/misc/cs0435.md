{  
   public class A { }  
}  
```  
  
 <span data-ttu-id="763da-110">Następnie Skompiluj ten plik:</span><span class="sxs-lookup"><span data-stu-id="763da-110">Then, compile this file:</span></span>

namespace Util
{  
   public class A { }  
}

{  
   public static void Main()
   {  
      Console.WriteLine(typeof(Util.A)); // CS0435  
   }  
}  
```
