 <span data-ttu-id="682c7-104">[Podstawowej](../../csharp/language-reference/keywords/base.md) użyto słowa kluczowego poza metoda, właściwość lub konstruktora.</span><span class="sxs-lookup"><span data-stu-id="682c7-104">The [base](../../csharp/language-reference/keywords/base.md) keyword was used outside of a method, property, or constructor.</span></span>

```csharp  
// CS1512.cs

class Base {}  
  
class CMyClass : Base  
{  
    private String xx = base.ToString();   // CS1512  
    // Try putting this initialization in the constructor instead:  
    // public CMyClass()  
    // {  
    //    xx = base.ToString();  
    // }  
  
    public static void Main()  
    {  
        CMyClass z = new CMyClass();  
    }  
}  
```
