```csharp

class x  
{  
   public static void Main()  
   {  
      int i = 6;  
  
      switch(i)   // CS1522  
      {  
         // add something to the switch block, for example:  
         /*  
         case (5):  
            Console.WriteLine("5");  
            return;  
         default:  
            Console.WriteLine("not 5");  
            return;  
         */  
      }  
   }  
}  
```
