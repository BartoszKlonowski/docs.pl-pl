## <a name="example"></a><span data-ttu-id="15d29-109">Przykład</span><span class="sxs-lookup"><span data-stu-id="15d29-109">Example</span></span>

using System;  
  
    public partial class C  
    {  
        // Defining declaration.  
        partial void Part();  
  
        // Implementing declaration.

        {  
            //...Do something.  
        }  
  
        // Second implementing declaration.  
        partial void Part() // CS0757  
        {  
            //...Do something.  
        }

        {  
            return 1;  
        }  
    }
