 <span data-ttu-id="23854-104">Zdarzenie w interfejsie nie może mieć inicjatora.</span><span class="sxs-lookup"><span data-stu-id="23854-104">An event in an interface cannot have an initializer.</span></span> <span data-ttu-id="23854-105">Aby uzyskać więcej informacji, zobacz [interfejsów](../../csharp/programming-guide/interfaces/index.md).</span><span class="sxs-lookup"><span data-stu-id="23854-105">For more information, see [Interfaces](../../csharp/programming-guide/interfaces/index.md).</span></span>

```csharp  
{  
   event MyDelegate d = new MyDelegate(M.f);   // CS0068

   // event MyDelegate d2;

{  
   event MyDelegate d = new MyDelegate(M.f);  
  
   public static void f()  
   {  
   }  
  
   public static void Main()  
   {  
   }  
}  
```
