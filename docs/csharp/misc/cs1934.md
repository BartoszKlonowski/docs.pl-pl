## <a name="example"></a><span data-ttu-id="c54b9-110">Przykład</span><span class="sxs-lookup"><span data-stu-id="c54b9-110">Example</span></span>

using System.Collections;  
static class Test  
{  
    public static void Main()  
    {

    }
