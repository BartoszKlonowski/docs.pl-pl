 <span data-ttu-id="d6d13-108">Poniższy przykład generuje CS0431:</span><span class="sxs-lookup"><span data-stu-id="d6d13-108">The following sample generates CS0431:</span></span>  
  
```csharp  
// CS0431.cs  
using A = Outer;  
  
public class Outer
{  
   public class Inner
   {  
      public static void Meth() {}  
   }  
}  
  
public class MyClass  
{  
   public static void Main()  
   {  
      A::Inner.Meth();   // CS0431  
      A.Inner.Meth();   // OK  
   }  
}  
```
