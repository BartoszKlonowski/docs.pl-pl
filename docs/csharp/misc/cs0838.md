## <a name="example"></a><span data-ttu-id="63b13-107">Przykład</span><span class="sxs-lookup"><span data-stu-id="63b13-107">Example</span></span>

using System.Linq;  
using System.Linq.Expressions;  
  
namespace TestNamespace  
{  
    class Test  
    {  
        static int Main()  
        {

            return 1;  
        }  
    }  
}
