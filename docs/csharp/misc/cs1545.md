## <a name="example"></a><span data-ttu-id="a33ed-107">Przykład</span><span class="sxs-lookup"><span data-stu-id="a33ed-107">Example</span></span>  
  
```cpp  
// CPP1545.cpp  
// compile with: /clr /LD  
// a Visual C++ program  
using namespace System;  
public ref struct Employee {  
   Employee( String^ s, int d ) {}  
  
   property String^ name {  
      String^ get() {  
         return nullptr;  
      }  
   }  
};  
  
public ref struct Manager {  
   property Employee^ Report [String^] {  
      Employee^ get(String^ s) {  
         return nullptr;  
      }  
  
      void set(String^ s, Employee^ e) {}  
   }  
};  
```  
  
## <a name="example"></a><span data-ttu-id="a33ed-108">Przykład</span><span class="sxs-lookup"><span data-stu-id="a33ed-108">Example</span></span>
