```csharp

{  
   public event del MyEvent  
   {  
      add;   // CS0073

      // {  
      //    MyEvent += value;  
      // }  
      remove  
      {  
         MyEvent -= value;  
      }  
  
   }  
  
   public static void Main()  
   {  
   }  
}  
```
