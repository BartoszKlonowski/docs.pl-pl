## <a name="example"></a><span data-ttu-id="44ca2-107">Przykład</span><span class="sxs-lookup"><span data-stu-id="44ca2-107">Example</span></span>

    public interface IF  
    {  
        void Part();  
    }  
  
    public partial class C : IF

        partial void IF.Part(); //CS0754

        {  
            return 1;  
        }  
    }
