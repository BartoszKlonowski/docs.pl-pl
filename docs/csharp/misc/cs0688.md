using System;  
using System.Security.Permissions;

{  
    //Uncomment the following line to close the security hole  
    //[FileIOPermission(SecurityAction.LinkDemand, All=@"C:\\")]

    {  
    }

{  
    [FileIOPermission(SecurityAction.LinkDemand, All=@"C:\\")] // CS0688

    {  
    }  
    static void Main()  
    {  
    }  
}
