 <span data-ttu-id="f37bf-105">Nie można użyć [abstrakcyjnej](../language-reference/keywords/abstract.md) klasy atrybutów niestandardowych jako atrybutu.</span><span class="sxs-lookup"><span data-stu-id="f37bf-105">An [abstract](../language-reference/keywords/abstract.md) custom attribute class cannot be used as an attribute.</span></span>

{  
}  
  
public class My2Attribute : MyAttribute
