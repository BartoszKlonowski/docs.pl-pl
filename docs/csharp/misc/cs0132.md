```csharp  
// CS0132.cs  
namespace MyNamespace  
{  
   public class MyClass  
   {  
      public MyClass(int i)  
      {  
      }  
   }  
  
   public class MyClass2 : MyClass  
   {  
      static MyClass2(int i)   // CS0132  
      {  
      }  
   }  
}  
```
