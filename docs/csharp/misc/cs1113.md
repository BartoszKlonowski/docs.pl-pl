## <a name="example"></a><span data-ttu-id="c2a75-110">Przykład</span><span class="sxs-lookup"><span data-stu-id="c2a75-110">Example</span></span>

public static class Extensions  
{  
    public static S ExtMethod(this S s)

    }

}
