 <span data-ttu-id="b88d0-108">Poniższy przykład generuje CS0419.</span><span class="sxs-lookup"><span data-stu-id="b88d0-108">The following sample generates CS0419.</span></span>  
  
```csharp  
// cs0419.cs  
// compile with: /doc:x.xml /W:3  
interface I  
{  
   /// text for F(void)  
   void F();  
   /// text for F(int)  
   void F(int i);  
}  
/// text for class MyClass  
public class MyClass  
{  
   /// <see cref="I.F"/>  
   public static void MyMethod(int i)  
   {  
   }  
/* Try this instead:  
   /// <see cref="I.F(int)"/>  
   public static void MyMethod(int i)  
   {  
   }  
*/  
   /// text for Main  
   public static void Main ()  
   {  
   }  
}  
```
