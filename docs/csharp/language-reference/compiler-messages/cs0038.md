<span data-ttu-id="ac898-103">Nie można uzyskać dostępu do niestatycznego elementu członkowskiego typu zewnętrznego "type1" za pośrednictwem zagnieżdżonego typu "type2"</span><span class="sxs-lookup"><span data-stu-id="ac898-103">Cannot access a nonstatic member of outer type 'type1' via nested type 'type2'</span></span>

```csharp

    public int count;
    // Try the following line instead.
    // public static int count;

    class InnerClass
    {
        void Func()
        {
            // or, create an instance
            // OuterClass class_inst = new OuterClass();
            // int count2 = class_inst.count;
            int count2 = count;   // CS0038
        }
    }

    public static void Main()
    {
    }
}
```
