 <span data-ttu-id="0ce02-104">Wykonano wywołanie do metody klasy, dla żadna definicja metody nie pobiera określonej liczby argumentów. </span><span class="sxs-lookup"><span data-stu-id="0ce02-104">A call was made to a class method, but no definition of the method takes the specified number of arguments.</span></span>  
  
## <a name="example"></a><span data-ttu-id="0ce02-105">Przykład</span><span class="sxs-lookup"><span data-stu-id="0ce02-105">Example</span></span>  
 <span data-ttu-id="0ce02-106">Poniższy przykład spowoduje wygenerowanie CS1501.</span><span class="sxs-lookup"><span data-stu-id="0ce02-106">The following sample generates CS1501.</span></span>  
  
```csharp  
using System;  
  
namespace ConsoleApplication1  
{  
    class Program  
    {  
        static void Main(string[] args)  
        {  
            ExampleClass ec = new ExampleClass();  
            ec.ExampleMethod();  
            ec.ExampleMethod(10);  
            // The following line causes compiler error CS1501 because   
            // ExampleClass does not contain an ExampleMethod that takes  
            // two arguments.  
            ec.ExampleMethod(10, 20);  
        }  
    }  
  
    // ExampleClass contains two overloads for ExampleMethod. One of them   
    // has no parameters and one has a single parameter.  
    class ExampleClass  
    {  
        public void ExampleMethod()  
        {  
            Console.WriteLine("Zero parameters");  
        }  
  
        public void ExampleMethod(int i)  
        {  
            Console.WriteLine("One integer parameter.");  
        }  
  
        //// To fix the error, you must add a method that takes two arguments.  
        //public void ExampleMethod (int i, int j)  
        //{  
        //    Console.WriteLine("Two integer parameters.");  
        //}  
    }  
}  
```
