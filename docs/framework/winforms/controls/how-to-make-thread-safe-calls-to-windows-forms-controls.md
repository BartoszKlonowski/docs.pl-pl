{
    thread2 = new Thread(new ThreadStart(WriteTextUnsafe));
    thread2.Start();
}
private void WriteTextUnsafe()

    textBox1.Text = "This text was set unsafely.";
}
```

    TextBox1.Text = "This text was set unsafely."
End Sub
```
