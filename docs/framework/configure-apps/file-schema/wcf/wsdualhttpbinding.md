## <a name="example"></a><span data-ttu-id="268cf-208">Przykład</span><span class="sxs-lookup"><span data-stu-id="268cf-208">Example</span></span>  
  
```xml  
<configuration>
  <system.ServiceModel>
    <bindings>
      <wsDualHttpBinding>
        <binding closeTimeout="00:00:10"
                 openTimeout="00:00:20"

                 useDefaultWebProxy="false">

                           inactivityTimeout="00:02:00" />

            <message clientCredentialType="None"
                     negotiateServiceCredential="false"
                     algorithmSuite="Aes128" />
          </security>
        </binding>
      </wsDualHttpBinding>
    </bindings>
  </system.ServiceModel>
</configuration>
```  
