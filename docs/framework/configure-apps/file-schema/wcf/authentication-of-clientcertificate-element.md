## <a name="example"></a><span data-ttu-id="1f9ce-192">Przykład</span><span class="sxs-lookup"><span data-stu-id="1f9ce-192">Example</span></span>

    <clientCertificate>
      <certificate findValue="www.cohowinery.com"

                      mapClientCertificateToWindowsAccount="true" />
