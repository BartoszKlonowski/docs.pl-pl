```csharp
class C
{
    public void M1()
    {
        // Violates rule CA1825.
        var a = new int[0];
    }
}
```

```csharp
class C
{
    public void M1()
    {
        // Resolves rule CA1825 violation.
        var a = System.Array.Empty<int>();
    }
```
