### <a name="violation"></a><span data-ttu-id="2459f-141">Krocz</span><span class="sxs-lookup"><span data-stu-id="2459f-141">Violation</span></span>

```csharp
using System;
using System.Security.Cryptography;

class ExampleClass
{
    public void ExampleMethod(string password, byte[] salt, int cb)
    {
        var iterations = 100;
        Random r = new Random();

        if (r.Next(6) == 4)
        {
            iterations = 100000;
        }

        var rfc2898DeriveBytes = new Rfc2898DeriveBytes(password, salt, iterations);
        rfc2898DeriveBytes.GetBytes(cb);
    }
}
```

### <a name="solution"></a><span data-ttu-id="2459f-142">Rozwiązanie</span><span class="sxs-lookup"><span data-stu-id="2459f-142">Solution</span></span>

```csharp
using System.Security.Cryptography;

class ExampleClass
{
    public void ExampleMethod(string password, byte[] salt, int cb)
    {
        var rfc2898DeriveBytes = new Rfc2898DeriveBytes(password, salt);
        rfc2898DeriveBytes.IterationCount = 100000;
        rfc2898DeriveBytes.GetBytes(cb);
    }
}
```
